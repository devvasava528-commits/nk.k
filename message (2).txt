<script>
    // ðŸ‘‡ CHANGE THIS TO YOUR LAPTOP IP
    const socket = io("http://YOUR-IP-HERE:5000");

    // Setup empty chart
    const ctx = document.getElementById('thdChart').getContext('2d');
    const thdChart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: [],       // time points
            datasets: [{
                label: 'THD %',
                data: [],
                borderWidth: 2,
                tension: 0.2,
                pointRadius: 0
            }]
        },
        options: {
            animation: false,
            scales: {
                x: {
                    title: { display: true, text: "Time (updates)" }
                },
                y: {
                    title: { display: true, text: "THD %" },
                    suggestedMin: 0
                }
            }
        }
    });

    socket.on("data_update", (data) => {
        // Update text
        document.getElementById("vf").innerText = data.Vf_RMS;
        document.getElementById("vd").innerText = data.Vd_RMS;
        document.getElementById("thd").innerText = data.THD;
        document.getElementById("thdpr").innerText = data.THD_pr + "%";

        // Update graph
        thdChart.data.labels.push("");   // simple x-axis point
        thdChart.data.datasets[0].data.push(data.THD_pr);

        // keep graph to last 30 points
        if (thdChart.data.labels.length > 30) {
            thdChart.data.labels.shift();
            thdChart.data.datasets[0].data.shift();
        }

        thdChart.update();
    });
</script>
